import ‘package:flutter/material.dart’;

class MyCallback extends StatefulWidget {
@override
_MyCallbackState createState() => _MyCallbackState();
}

class _MyCallbackState extends State {
var _myMsg;
@override
Widget build(BuildContext context) {
return Scaffold(
appBar: AppBar(title: Text(“MyCallback”)),
body: Center(
child:Column(
children: [
RaisedButton(
child: Text(‘MyCallback’),
onPressed: () {
Navigator.push(context,
MaterialPageRoute(builder: (BuildContext context) {
return MyNewCallback(
param: ‘Hello’,
callBack: (String msg) {
print(msg);
setState(() {
_myMsg = msg;
});
});
}));
},
),
Text(“$_myMsg”),
],
),
),
);
}
}

class MyNewCallback extends StatelessWidget {
final String param; // home过来的参数
final Function callBack; // 回调方法
MyNewCallback({Key key, this.param, this.callBack}) : super(key: key);

_printParam() {
  print(this.param);
}

@override
Widget build(BuildContext context) {
  _printParam();
  return Scaffold(
    appBar: AppBar(title: Text('MyNewCallback')),
    body: Center(
      child: Column(
        children: <Widget>[
          RaisedButton(
            onPressed: () {
              this.callBack('World');
              Navigator.of(context).pop();
            },
            child: Text('返回'),
          ),
          Text("$param"),
        ],
      ),
    )
  );
}

}
