class SortTest {
public SortTest() {}
// 冒泡排序 时间复杂度 O(n2)n的平方
int[] bubbleSort(int[] sortArray) {
for (int i=0 ; i<sortArray.length ; i++) {
for (int j=i+1 ; j<sortArray.length ; j++) {
if (sortArray[i] > sortArray[j]) {
int tmpValue = sortArray[i];
sortArray[i] = sortArray[j];
sortArray[j] = tmpValue;
}
}
}
return sortArray;
}
// 冒泡排序 方法2
int[] bubbleSort2(int[] sortArray) {
for (int i=0 ; i<sortArray.length ; i++) {
for (int j=0 ; j<sortArray.length-1-i ; j++) {
if (sortArray[j] > sortArray[j+1]) {
int tmpValue = sortArray[j];
sortArray[j] = sortArray[j+1];
sortArray[j+1] = tmpValue;
}
}
}
return sortArray;
}
// 选择排序 时间复杂度 O(n2)n的平方
int[] selectSort(int[] sortArray) {
for (int i=0 ; i<sortArray.length ; i++) {
int minIndex = i;
for (int j=i+1; j< sortArray.length ; j++) {
if (sortArray[minIndex] > sortArray[j]) {
minIndex = j;
}
}
if (i != minIndex) {
int tmpValue = sortArray[i];
sortArray[i] = sortArray[minIndex];
sortArray[minIndex] = tmpValue;
}
}
return sortArray;
}
// 插入排序 时间复杂度 O(n2)n的平方
int[] insertSort(int[] sortArray) {
for (int i=1 ; i<sortArray.length ; i++) {
int tmpValue = sortArray[i];
int j = i - 1;
while (j >= 0 && sortArray[j] > tmpValue) {
sortArray[j+1] = sortArray[j];
j–;
}
sortArray[j+1] = tmpValue;
}
return sortArray;
}
// 快速排序,二分排序,折半排序 时间复杂度 O(nlogn)
int[] quicksort(int sortArray[],int start,int end) {
int midValue = sortArray[start];
int left = start;
int right = end;
while (left < right) {
while ((left < right) && (sortArray[right] > midValue)) {
right–;
}
while ((left < right) && (sortArray[left] < midValue)) {
left++;
}
if ((sortArray[left] == sortArray[right]) && (left < right)) {
left++;
} else {
int temp = sortArray[left];
sortArray[left] = sortArray[right];
sortArray[right] = temp;
}
}
if (left - 1 > start) sortArray = quicksort(sortArray, start, left - 1);
if (right + 1 < end) sortArray = quicksort(sortArray, right + 1, end);
return sortArray;
}
// 希尔排序 缩小增量排序 时间复杂度 O(n2)n的平方
int[] shellSort(int[] sortArray) {
int gap = sortArray.length/2;
while (gap >= 1) {
int i = gap;
while (i < sortArray.length) {
int tmpValue = sortArray[i];
int j = i - gap;
while (j >= 0 && tmpValue < sortArray[j]) {
sortArray[j+gap] = sortArray[j];
j = j - gap;
}
if (j != (i - gap)) {
sortArray[j+gap] = tmpValue;
}
i++;
}
gap = gap/2;
}
return sortArray;
}
// 归并排序 时间复杂度 O(nlogn)
void mergeSort(int[] arr, int[] result, int lefts, int rights) {
if (lefts >= rights)
return;
int mid = (rights - lefts)/2 + lefts;
int left1 = lefts, right1 = mid;
int left2 = mid + 1, right2 = rights;
mergeSort(arr, result, left1, right1);
mergeSort(arr, result, left2, right2);
int k = lefts;
while (left1 <= right1 && left2 <= right2)
result[k++] = arr[left1] < arr[left2] ? arr[left1++] : arr[left2++];
while (left1 <= right1)
result[k++] = arr[left1++];
while (left2 <= right2)
result[k++] = arr[left2++];
for (k = lefts; k <= rights; k++)
arr[k] = result[k];
}
}
