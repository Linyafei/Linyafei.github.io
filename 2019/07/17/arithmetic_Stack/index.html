class StackArray {
String[] dataArray;
int size = 0;
public StackArray(int capacity) {
this.dataArray = new String[capacity];
}
public void display() {
for (int i=0;i<this.dataArray.length;i++) {
System.out.println(“==” + this.dataArray[i]);
}
}
public void push(String data) {
if (size == 0) {
this.dataArray[0] = data;
}else{
for (int i=size;i>0;i–) {
this.dataArray[i] = this.dataArray[i-1];
}
this.dataArray[0] = data;
}
size ++;
}
public void pop() {
for (int i=0;i<size;i++) {
this.dataArray[i] = this.dataArray[i+1];
}
this.dataArray[size-1] = null;
size –;
}
}

class StackLinkList {
class Node {
String data;
Node nextNode;
public Node() {}
public Node(String data) {
this.data = data;
}
}
public Node headNode;
public StackLinkList() {
this.headNode = new Node();
}
public void display() {
Node tmpNode = headNode;
while (tmpNode.nextNode != null) {
System.out.print(“==” + tmpNode.data);
tmpNode = tmpNode.nextNode;
}
}
public void push(String data) {
Node tmpNode = new Node(data);
tmpNode.nextNode = headNode;
headNode = tmpNode;
}
public void pop() {
Node tmpNode = new Node();
tmpNode = headNode.nextNode;
headNode = tmpNode;
}
public Boolean isEmpty() {
if (headNode == null) {
return true;
}else {
return false;
}
}
public int size() {
if (headNode != null) {
Node tmpNode = headNode;
int tmpSize = 1;
while (tmpNode.nextNode != null) {
tmpNode = tmpNode.nextNode;
tmpSize ++;
}
return tmpSize;
}else {
return -1;
}
}
}
