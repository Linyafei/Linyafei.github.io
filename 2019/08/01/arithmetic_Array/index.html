class MyArray {
private ElementType[] dataArray;
private int size = 0;
//
public MyArray(int capacity) {
this.dataArray = (ElementType[]) new Object[capacity];
}
public MyArray() {
new MyArray<>(10);
}
public int getCapacity() {
return dataArray.length;
}
public int getSize() {
return size;
}
public boolean isEmpty() {
return size == 0;
}
public void add(int index,ElementType e) {
if(index < 0 || index > size) {
throw new IllegalArgumentException(“越界”);
}
if(index == size){
resize(2 * dataArray.length);
}
for (int i = size - 1; i >= index; i–) {
dataArray[i+1] = dataArray[i];
}
dataArray[index] = e;
size ++;
}
public void addLast(ElementType e) {
add(size,e);
}
public void addFirst(ElementType e) {
add(0,e);
}
public boolean contains(ElementType e) {
for (ElementType tmpE:dataArray) {
if(tmpE.equals(e)) {
return true;
}
}
return false;
}
private void resize(int newCapacity) {
ElementType[] newData = (ElementType[]) new Object[newCapacity];
for (int i=0; i < size; i++) {
newData[i] = dataArray[i];
}
dataArray = newData;
}
public int find(ElementType e) {
for (int i = 0; i < dataArray.length; i++) {
if(dataArray[i].equals(e)) {
return i;
}
}
return -1;
}
public int removeElement(ElementType e) {
int index = find(e);
if(index != -1) {
remove(index);
}
return index;
}
public ElementType remove(int index) {
if(index < 0 || index >= size) {
throw new IllegalArgumentException(“越界”);
}
ElementType removeData = dataArray[index];
for (int i = index + 1; i < size; i++) {
dataArray[i-1] = dataArray[i];
}
dataArray[index] = null;
size – ;
if(size <= dataArray.length/4 && dataArray.length/2 != 0) {
resize(dataArray.length/2);
}
return removeData;
}
//列出所有元素
public void display(){
for(int i=0;i<size;i++){
System.out.print(dataArray[i]+”\t”);
}
}
}
