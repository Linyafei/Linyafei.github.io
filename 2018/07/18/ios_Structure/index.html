  • (NSMutableArray *)modelsWithOrganizationModel:(StructureModel *)model index:(NSInteger)index
    {
    NSMutableArray *arrayM = [[NSMutableArray alloc]init];
    [arrayM addObject:model];

    StructureModel * previousNode = nil;
    for (StructureModel *subModel in model.detail) {

      subModel.isClose = NO;
  subModel.index = index + 1;
  subModel.previousNode = previousNode;
  subModel.parentNode = model;
  NSMutableArray * subArryM = [self modelsWithOrganizationModel:subModel index:index + 1];
  [arrayM addObjectsFromArray:subArryM];
  previousNode = subModel;
  subModel.previousNode.nextNode = subModel;

    }

    return arrayM;
    }

  • (UITableViewCell *)tableView:(UITableView *)tableView cellForRowAtIndexPath:(NSIndexPath *)indexPath
    {
    StructureModel *model = [[StructureModel alloc]init];
    model = _allDataArrM[indexPath.row];

    NSString *cellId = @”cellId”;
    StructureCell *cell = [tableView dequeueReusableCellWithIdentifier:cellId];
    if (!cell) {

      cell = [[StructureCell alloc]initWithStyle:UITableViewCellStyleDefault reuseIdentifier:cellId];

    }
    cell.selectionStyle = UITableViewCellAccessoryNone;

    [cell setSModel:model];
    return cell;
    }

  • (void)tableView:(UITableView *)tableView didSelectRowAtIndexPath:(NSIndexPath *)indexPath
    {
    StructureModel *model = [[StructureModel alloc]init];
    model = _allDataArrM[indexPath.row];

    if (model.detail.count == 0) {

      return;

    }

    if (model.isClose) {

      NSMutableArray *temArrIndexM = [[NSMutableArray alloc]init];
  temArrIndexM = [self removeObjectsInArray:model];
  [_allDataArrM removeObjectsInArray:temArrIndexM];

    }else{

      for (NSInteger i=0; i<model.detail.count; i++) {
      StructureModel *subModel = model.detail[i];
      [_allDataArrM insertObject:subModel atIndex:indexPath.row + 1 + i];
  }

    }
    model.isClose = !model.isClose;
    [_tableView reloadData];
    }

  • (NSMutableArray *)removeObjectsInArray:(StructureModel *)model
    {
    if (model.detail.count == 0) {

      return [NSMutableArray new];

    }
    NSMutableArray *temArrIndexM = [[NSMutableArray alloc]init];
    for (StructureModel *subModel in model.detail) {

      [temArrIndexM addObject:subModel];
  NSMutableArray * subArryM = [self removeObjectsInArray:subModel];
  [temArrIndexM addObjectsFromArray:subArryM];

    }
    return temArrIndexM;
    }
